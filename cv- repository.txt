        async function callGemini(systemPrompt, userQuery, outputElement) {
            outputElement.classList.remove('hidden');
            outputElement.innerHTML = `<p class="text-center text-gray-500">Generating...</p>`;
            
            try {
                const payload = {
                    contents: [{ parts: [{ text: userQuery }] }],
                    systemInstruction: { parts: [{ text: systemPrompt }] },
                };

                const response = await fetch(API_URL + API_KEY, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                
                if (!response.ok) {
                    throw new Error(`API call failed with status: ${response.status}`);
                }

                const result = await response.json();
                const text = result.candidates?.[0]?.content?.parts?.[0]?.text;

                if (text) {
                    outputElement.innerHTML = `<p class="text-gray-700 whitespace-pre-wrap">${text}</p>`;
                } else {
                    outputElement.innerHTML = `<p class="text-red-500">Sorry, something went wrong. Please try again.</p>`;
                }
            } catch (error) {
                console.error("Error during API call:", error);
                outputElement.innerHTML = `<p class="text-red-500">Sorry, something went wrong. Please try again.</p>`;
            }
        }
